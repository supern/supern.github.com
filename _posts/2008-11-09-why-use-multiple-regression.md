    #png(width = 500, height = 400)
    set.seed(711)
    x = seq(0, 4, length = 100)
    y = -x + jitter(rep(1:5, each = 20), 2)
    z = rep(1:5, each = 20)
    #par(mar = c(4, 4, 0.1, 0.1))
    plot(x, y)
    abline(lm(y ~ x), col = "red")
    plot(x, y, pch = 20, col = z)
    lines(lowess(x, y, f = 0.1), col = "red")
    summary(lm(y ~ x))
    summary(lm(y ~ x + z))
    #dev.off()


注意本例中局部平滑函数lowess()函数的应用。
